You are given a **sorted** array and a target value. Find the **first index** where the value is **greater than or equal to** the target.

This is exactly like `std::lower_bound()` in C++ STL.

  public:
    int lowerBound(vector<int>& arr, int target) {
        int low = 0;                         // ğŸ”½ Start of the array
        int high = arr.size() - 1;           // ğŸ”¼ End of the array
        int lower_bound = arr.size();        // ğŸ“¦ Default to size() in case target > all elements

            int mid = low + (high - low) / 2; // ğŸ§® Avoid overflow

            if (arr[mid] >= target) {
                // ğŸ¯ Found a value >= target
                lower_bound = mid;           // âœ… Store potential answer
                high = mid - 1;              // ğŸ” Search left for earlier answer
            } else {
                // âŒ Too small, go right
                low = mid + 1;
            }
        }

        return lower_bound;  // ğŸ“¬ Return the index where target should go
    }
};
